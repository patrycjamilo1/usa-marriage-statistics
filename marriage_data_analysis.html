5. Zaleno midzy dochodem rodziny a iloci dzieci

Tworzenie nowej ramki danych z dwoma zmiennymi: dochd rodziny i ilosc dzieci

dochod_i_dzieci <- data.frame(Dochod_Rodziny = data$income, Ilosc_Dzieci = data$child6 + data$child618)

Wykres liniowy (geom_smooth: dodanie linii regresji, geom_point: dodanie punktw danych(alpha - przezroczystosc punktow)) lm - liniowa regresja, se = FALSE - bez prawdopobienstwa linii regresji, dzieki czemu wykres jest bardziej przejerzysty)

ggplot(dochod_i_dzieci, aes(x = Dochod_Rodziny, y = Ilosc_Dzieci)) +
  geom_smooth(method = "lm", se = FALSE, color = "red") +
  geom_point(color = "gray", size = 3, alpha = 0.5) +
  labs(title = "Zaleno midzy dochodem rodziny a iloci dzieci",
       x = "Dochd rodziny",
       y = "Ilo dzieci") +
  theme_minimal() +
  theme(plot.title = element_text(hjust = 0.5))
## `geom_smooth()` using formula = 'y ~ x'
